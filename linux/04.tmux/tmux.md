# bind
    bind指的是Ctrl + A
# 窗格(panes)操作
    bind + -        #水平分割窗格
    bind + _        #竖直分割窗格
    bind + z        #当前窗格最大化/取消最大化
    bind + +        #当前窗格最大化/取消最大化
    bind + h/j/k/l  #切换窗格
    bind + H/J/K/L  #重新定义窗格大小
# 窗口(Window)操作
    bind + c        #创建窗口
    bind + ,        #重命名当前窗口
    bind + w        #选择窗口
    bind + 数字     #切换窗口，比如bind +2 切换到窗口2
    bind + &        杀死当前窗口
# 会话(session)操作
    bind + s        #选择会话
    bind + C-c      #创建会话
    bind + $        #重命名当前会话
